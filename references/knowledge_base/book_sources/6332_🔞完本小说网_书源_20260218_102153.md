function sortChapters(data) {
  // 检测是否全局严格递增
  const isStrictlyIncreasing = (arr, idx) => {
    for (let i = 1; i < arr.length; i++) {
      if (arr[i].intValues[idx] <= arr[i-1].intValues[idx]) {
        return false;
      }
    }
    return true;
  };

  // 确定排序索引：递增列是显示序号，另一列用于排序
  let sortIndex;
  if (isStrictlyIncreasing(data, 0)) {
    sortIndex = 1; // 第0列递增，按第1列排序
  } else if (isStrictlyIncreasing(data, 1)) {
    sortIndex = 0; // 第1列递增，按第0列排序
  } else {
    // 兜底：如果都不递增，默认按第1列排序
    sortIndex = 1;
  }

  // 排序并移除 intValues
  return data
